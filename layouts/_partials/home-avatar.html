{{- /* Renders one contributor avatar for the home marquee. Params: page (contributor Page), dup (bool). The marquee renders each row twice for a seamless CSS loop; the dup copies are non-link s with empty alt, so text browsers (lynx/w3m) & screen readers list each contributor only once — no doubled links. */ -}} {{- $p := .page -}} {{- $dup := $.dup -}} {{- $images := $p.Resources.ByType "image" -}} {{- $avatar := partial "utils/contributor-avatar.html" (dict "images" $images "patterns" (slice "avatar_card*" "avatar_single*" "*feature*")) -}} {{- if $dup -}} {{ with $avatar }}{{ $sm := .Resize "120x webp q80" }}{{ else }}{{ substr $p.Title 0 1 | upper }}{{ end }} {{- else -}} {{ with $avatar }}{{ $sm := .Resize "120x webp q80" }}{{ else }}{{ substr $p.Title 0 1 | upper }}{{ end }} {{- end -}}
